Stars

Projekt kus õppisime nested loope kasutama ja printisime *'idest mustri. 

        
            using System;
            using System.Collections.Generic;
            using System.Linq;
            using System.Text;
            using System.Threading.Tasks;

            namespace ProgTask2
            {
                static class Stars
                {
                    public static void PrintCharacters() // ehk nested loop
                    {
                        Console.WriteLine();
                        for (int i = 7; i >= 1 ; i--) 
                        {
                            for (int j = 1; j <= i; j++)
                            {
                                Console.Write("*");
                            }
                            Console.WriteLine();
                        }

                    }
                }
            }
